function calculateAnnual(inputId, frequencyId, resultId) { const input = document.getElementById(inputId).value; const frequency = document.getElementById(frequencyId).value; let annualValue; switch (frequency) { case 'annually': annualValue = input; break; case 'quarterly': annualValue = input * 4; break; case 'monthly': annualValue = input * 12; break; case 'weekly': annualValue = input * 52; break; default: annualValue = ''; } document.getElementById(resultId).textContent = ` (Equivalent to $${annualValue} annually)`; }
